/**
 * Transistor - Simple Radio App for HarmonyOS
 * Main Page - Station List and Player
 */
import router from '@ohos.router';
import promptAction from '@ohos.promptAction';
import hilog from '@ohos.hilog';
import { Station, StationCollection, PLAYBACK_STATE_STOPPED, PLAYBACK_STATE_LOADING, PLAYBACK_STATE_PLAYING } from '../model/Station';
import { AudioPlayerService } from '../service/AudioPlayerService';

const TAG = 'Index';
const DOMAIN = 0x0000;

@Entry
@Component
struct Index {
  @State private stations: Station[] = [];
  @State private currentStation: Station | null = null;
  @State private isPlaying: boolean = false;
  @State private playbackState: number = PLAYBACK_STATE_STOPPED;
  @State private showAddDialog: boolean = false;
  @State private newStationName: string = '';
  @State private newStationUrl: string = '';
  @State private isPlayerExpanded: boolean = false;
  @State private showDeleteConfirm: boolean = false;
  @State private stationToDelete: Station | null = null;

  private audioService: AudioPlayerService = AudioPlayerService.getInstance();
  private stationCollection: StationCollection = StationCollection.getInstance();

  aboutToAppear() {
    hilog.info(DOMAIN, TAG, 'Index page aboutToAppear');
    this.loadStations();
    this.setupPlayerCallbacks();
  }

  /**
   * Load stations from storage
   */
  loadStations() {
    this.stationCollection.loadFromStorage();
    this.stations = this.stationCollection.getStations();
    if (this.stations.length > 0 && !this.currentStation) {
      this.currentStation = this.stations[0];
    }
    hilog.info(DOMAIN, TAG, 'Loaded %{public}d stations', this.stations.length);
  }

  /**
   * Setup player callbacks
   */
  setupPlayerCallbacks() {
    this.audioService.setPlaybackStateCallback((state: number, station: Station) => {
      this.playbackState = state;
      this.isPlaying = state === PLAYBACK_STATE_PLAYING;
      this.currentStation = station;
    });

    this.audioService.setErrorCallback((error: string) => {
      promptAction.showToast({ message: error, duration: 3000 });
    });

    // Initialize audio service
    this.audioService.init().catch((e) => {
      hilog.error(DOMAIN, TAG, 'Failed to init audio service: %{public}s', JSON.stringify(e));
    });
  }

  /**
   * Toggle playback
   */
  async togglePlayback() {
    if (!this.currentStation) {
      promptAction.showToast({ message: 'No station selected', duration: 2000 });
      return;
    }

    if (this.isPlaying) {
      await this.audioService.stop();
    } else {
      await this.audioService.play(this.currentStation);
    }
  }

  /**
   * Play a specific station
   */
  async playStation(station: Station) {
    this.currentStation = station;
    await this.audioService.play(station);
    this.isPlayerExpanded = false;
  }

  /**
   * Add new station
   */
  async addStation() {
    if (!this.newStationName.trim() || !this.newStationUrl.trim()) {
      promptAction.showToast({ message: 'Please fill in all fields', duration: 2000 });
      return;
    }

    if (!this.newStationUrl.startsWith('http://') && !this.newStationUrl.startsWith('https://')) {
      promptAction.showToast({ message: 'Invalid URL format', duration: 2000 });
      return;
    }

    const newStation = new Station(this.newStationName.trim(), this.newStationUrl.trim());
    newStation.playlistContent = newStation.createM3UContent();

    if (this.stationCollection.addStation(newStation)) {
      this.loadStations();
      promptAction.showToast({ message: 'Station added', duration: 2000 });
      this.showAddDialog = false;
      this.newStationName = '';
      this.newStationUrl = '';
    } else {
      promptAction.showToast({ message: 'Station already exists', duration: 2000 });
    }
  }

  /**
   * Delete station
   */
  deleteStation() {
    if (this.stationToDelete) {
      this.stationCollection.removeStation(this.stationToDelete);
      this.loadStations();
      if (this.currentStation?.id === this.stationToDelete.id) {
        this.currentStation = this.stations.length > 0 ? this.stations[0] : null;
      }
      promptAction.showToast({ message: 'Station deleted', duration: 2000 });
    }
    this.showDeleteConfirm = false;
    this.stationToDelete = null;
  }

  /**
   * Get playback button icon
   */
  getPlaybackIcon(): string {
    switch (this.playbackState) {
      case PLAYBACK_STATE_LOADING:
        return 'loading';
      case PLAYBACK_STATE_PLAYING:
        return 'stop';
      default:
        return 'play';
    }
  }

  /**
   * Get playback button text
   */
  getPlaybackText(): string {
    switch (this.playbackState) {
      case PLAYBACK_STATE_LOADING:
        return 'Loading...';
      case PLAYBACK_STATE_PLAYING:
        return 'Stop';
      default:
        return 'Play';
    }
  }

  /**
   * Build station list item
   */
  @Builder
  StationListItem(station: Station) {
    ListItem() {
      Row() {
        Column() {
          Image(station.imageUrl || $r('app.media.ic_launcher'))
            .width(56)
            .height(56)
            .borderRadius(8)
            .alt($r('app.media.ic_launcher'))
        }
        .width(72)
        .height(56)
        .alignItems(HorizontalAlign.Start)

        Column() {
          Text(station.name)
            .fontSize(16)
            .fontWeight(FontWeight.Medium)
            .fontColor('#212121')
            .maxLines(1)
            .textOverflow({ overflow: TextOverflow.Ellipsis })

          if (station.metadata) {
            Text(station.metadata)
              .fontSize(14)
              .fontColor('#757575')
              .maxLines(1)
              .textOverflow({ overflow: TextOverflow.Ellipsis })
          }
        }
        .layoutWeight(1)
        .padding({ left: 12 })
        .alignItems(HorizontalAlign.Start)

        Column() {
          if (this.currentStation?.id === station.id && this.isPlaying) {
            LoadingProgress()
              .width(24)
              .height(24)
              .color('#6200EE')
          }
        }
        .width(48)
        .height(48)
        .alignItems(HorizontalAlign.Center)
        .justifyContent(FlexAlign.Center)
      }
      .width('100%')
      .height(72)
      .padding({ left: 16, right: 16 })
      .backgroundColor(this.currentStation?.id === station.id ? '#F5F5F5' : '#FFFFFF')
    }
    .onClick(() => {
      this.playStation(station);
    })
    .onLongPress(() => {
      this.stationToDelete = station;
      this.showDeleteConfirm = true;
    })
  }

  build() {
    Stack() {
      Column() {
        // App Bar
        Row() {
          Text('Transistor')
            .fontSize(20)
            .fontWeight(FontWeight.Bold)
            .fontColor('#212121')

          Blank()

          Button('+')
            .fontSize(24)
            .fontWeight(FontWeight.Bold)
            .backgroundColor('#6200EE')
            .width(48)
            .height(48)
            .borderRadius(24)
            .onClick(() => {
              this.showAddDialog = true;
            })
        }
        .width('100%')
        .height(56)
        .padding({ left: 16, right: 16 })
        .backgroundColor('#FFFFFF')

        Divider().color('#BDBDBD')

        // Station List
        if (this.stations.length === 0) {
          Column() {
            Text('No stations added yet')
              .fontSize(16)
              .fontColor('#757575')
              .margin({ top: 100 })

            Text('Tap + to add a radio station')
              .fontSize(14)
              .fontColor('#9E9E9E')
              .margin({ top: 8 })
          }
          .width('100%')
          .height('100%')
          .alignItems(HorizontalAlign.Center)
        } else {
          List() {
            ForEach(this.stations, (station: Station) => {
              this.StationListItem(station)
            })
          }
          .width('100%')
          .layoutWeight(1)
          .divider({ strokeWidth: 1, color: '#E0E0E0', startMargin: 88, endMargin: 16 })
        }
      }
      .width('100%')
      .height('100%')
      .backgroundColor('#FFFFFF')

      // Mini Player
      if (this.currentStation) {
        Column() {
          if (this.isPlayerExpanded) {
            // Expanded Player
            Column() {
              Row() {
                Image(this.currentStation.imageUrl || $r('app.media.ic_launcher'))
                  .width(200)
                  .height(200)
                  .borderRadius(16)
                  .alt($r('app.media.ic_launcher'))
              }
              .width('100%')
              .height(200)
              .justifyContent(FlexAlign.Center)

              Text(this.currentStation.name)
                .fontSize(20)
                .fontWeight(FontWeight.Bold)
                .fontColor('#212121')
                .margin({ top: 16 })

              Text(this.currentStation.metadata || '')
                .fontSize(14)
                .fontColor('#757575')
                .margin({ top: 8 })

              Row() {
                Button('Play/Stop')
                  .fontSize(16)
                  .backgroundColor('#6200EE')
                  .width(120)
                  .height(48)
                  .borderRadius(24)
                  .onClick(() => {
                    this.togglePlayback();
                  })
              }
              .margin({ top: 24 })

              // Station Details
              Column() {
                this.DetailRow('Stream URL', this.currentStation.streamUri)
                this.DetailRow('MIME Type', this.currentStation.mimeType || 'N/A')
                this.DetailRow('Channels', this.currentStation.channelCount > 0 ? String(this.currentStation.channelCount) : 'N/A')
                this.DetailRow('Sample Rate', this.currentStation.sampleRate > 0 ? String(this.currentStation.sampleRate) : 'N/A')
              }
              .width('100%')
              .padding(16)
              .margin({ top: 16 })
            }
            .width('100%')
            .height('100%')
            .backgroundColor('#FFFFFF')
            .padding(16)
          } else {
            // Mini Player Bar
            Row() {
              Image(this.currentStation.imageUrl || $r('app.media.ic_launcher'))
                .width(48)
                .height(48)
                .borderRadius(8)
                .alt($r('app.media.ic_launcher'))

              Column() {
                Text(this.currentStation.name)
                  .fontSize(14)
                  .fontWeight(FontWeight.Medium)
                  .fontColor('#212121')
                  .maxLines(1)
                  .textOverflow({ overflow: TextOverflow.Ellipsis })

                Text(this.currentStation.metadata || '')
                  .fontSize(12)
                  .fontColor('#757575')
                  .maxLines(1)
                  .textOverflow({ overflow: TextOverflow.Ellipsis })
              }
              .layoutWeight(1)
              .padding({ left: 12 })
              .alignItems(HorizontalAlign.Start)

              Button(this.getPlaybackText())
                .fontSize(14)
                .backgroundColor('#6200EE')
                .height(40)
                .borderRadius(20)
                .onClick(() => {
                  this.togglePlayback();
                })
            }
            .width('100%')
            .height(64)
            .padding({ left: 16, right: 16 })
            .backgroundColor('#F5F5F5')
            .onClick(() => {
              this.isPlayerExpanded = true;
            })
          }
        }
        .width('100%')
        .alignItems(HorizontalAlign.End)
      }

      // Add Station Dialog
      if (this.showAddDialog) {
        Column() {
          Column() {
            Text('Add Station')
              .fontSize(20)
              .fontWeight(FontWeight.Bold)
              .fontColor('#212121')

            TextInput({ placeholder: 'Station Name' })
              .width('100%')
              .height(48)
              .margin({ top: 16 })
              .onChange((value: string) => {
                this.newStationName = value;
              })

            TextInput({ placeholder: 'Stream URL (http://...)' })
              .width('100%')
              .height(48)
              .margin({ top: 12 })
              .onChange((value: string) => {
                this.newStationUrl = value;
              })

            Row() {
              Button('Cancel')
                .fontSize(16)
                .backgroundColor('#9E9E9E')
                .width('45%')
                .height(48)
                .onClick(() => {
                  this.showAddDialog = false;
                  this.newStationName = '';
                  this.newStationUrl = '';
                })

              Button('Add')
                .fontSize(16)
                .backgroundColor('#6200EE')
                .width('45%')
                .height(48)
                .onClick(() => {
                  this.addStation();
                })
            }
            .width('100%')
            .margin({ top: 24 })
            .justifyContent(FlexAlign.SpaceBetween)
          }
          .width('90%')
          .padding(24)
          .backgroundColor('#FFFFFF')
          .borderRadius(16)
        }
        .width('100%')
        .height('100%')
        .backgroundColor('#80000000')
        .justifyContent(FlexAlign.Center)
        .alignItems(VerticalAlign.Center)
      }

      // Delete Confirmation Dialog
      if (this.showDeleteConfirm) {
        Column() {
          Column() {
            Text('Delete Station')
              .fontSize(20)
              .fontWeight(FontWeight.Bold)
              .fontColor('#212121')

            Text('Are you sure you want to delete this station?')
              .fontSize(14)
              .fontColor('#757575')
              .margin({ top: 12 })

            Row() {
              Button('Cancel')
                .fontSize(16)
                .backgroundColor('#9E9E9E')
                .width('45%')
                .height(48)
                .onClick(() => {
                  this.showDeleteConfirm = false;
                  this.stationToDelete = null;
                })

              Button('Delete')
                .fontSize(16)
                .backgroundColor('#D32F2F')
                .width('45%')
                .height(48)
                .onClick(() => {
                  this.deleteStation();
                })
            }
            .width('100%')
            .margin({ top: 24 })
            .justifyContent(FlexAlign.SpaceBetween)
          }
          .width('85%')
          .padding(24)
          .backgroundColor('#FFFFFF')
          .borderRadius(16)
        }
        .width('100%')
        .height('100%')
        .backgroundColor('#80000000')
        .justifyContent(FlexAlign.Center)
        .alignItems(VerticalAlign.Center)
      }
    }
    .width('100%')
    .height('100%')
  }

  @Builder
  DetailRow(label: string, value: string) {
    Row() {
      Text(label)
        .fontSize(14)
        .fontColor('#757575')
        .width(100)

      Text(value)
        .fontSize(14)
        .fontColor('#212121')
        .layoutWeight(1)
    }
    .width('100%')
    .padding({ top: 4, bottom: 4 })
  }
}
