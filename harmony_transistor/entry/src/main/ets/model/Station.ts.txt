/**
 * Transistor - Simple Radio App for HarmonyOS
 * Station data model
 */

// 播放状态常量
export const PLAYBACK_STATE_STOPPED = 0;
export const PLAYBACK_STATE_LOADING = 1;
export const PLAYBACK_STATE_PLAYING = 2;

/**
 * Station class - represents a radio station
 */
export class Station {
  id: string;
  name: string;
  streamUri: string;
  imageUrl: string;
  playlistContent: string;
  metadata: string;
  mimeType: string;
  channelCount: number;
  sampleRate: number;
  bitrate: number;
  playbackState: number;
  selected: boolean;
  createdAt: number;
  imageSize: number;

  constructor(
    name: string = '',
    streamUri: string = '',
    imageUrl: string = '',
    playlistContent: string = ''
  ) {
    this.id = this.generateId(name, streamUri);
    this.name = name;
    this.streamUri = streamUri;
    this.imageUrl = imageUrl;
    this.playlistContent = playlistContent;
    this.metadata = '';
    this.mimeType = '';
    this.channelCount = 0;
    this.sampleRate = 0;
    this.bitrate = 0;
    this.playbackState = PLAYBACK_STATE_STOPPED;
    this.selected = false;
    this.createdAt = Date.now();
    this.imageSize = 0;
  }

  /**
   * Generate unique ID from station name and stream URI
   */
  private generateId(name: string, streamUri: string): string {
    return `${name.toLowerCase()}_${streamUri.hashCode()}`;
  }

  /**
   * Create M3U playlist content
   */
  createM3UContent(): string {
    return `#EXTM3U\n\n#EXTINF:-1,${this.name}\n${this.streamUri}\n`;
  }

  /**
   * Parse M3U playlist content
   */
  static parseM3U(content: string): { name: string; uri: string }[] {
    const results: { name: string; uri: string }[] = [];
    const lines = content.split('\n');

    let currentName = '';

    for (let i = 0; i < lines.length; i++) {
      const line = lines[i].trim();

      if (line.startsWith('#EXTINF:')) {
        const match = line.match(/,#(.+)$/);
        if (match) {
          currentName = match[1];
        }
      } else if (line.startsWith('http')) {
        results.push({
          name: currentName || 'Unknown Station',
          uri: line
        });
        currentName = '';
      }
    }

    return results;
  }

  /**
   * Create station from JSON object
   */
  static fromJSON(json: any): Station {
    const station = new Station();
    station.id = json.id || '';
    station.name = json.name || '';
    station.streamUri = json.streamUri || '';
    station.imageUrl = json.imageUrl || '';
    station.playlistContent = json.playlistContent || '';
    station.metadata = json.metadata || '';
    station.mimeType = json.mimeType || '';
    station.channelCount = json.channelCount || 0;
    station.sampleRate = json.sampleRate || 0;
    station.bitrate = json.bitrate || 0;
    station.playbackState = json.playbackState || PLAYBACK_STATE_STOPPED;
    station.selected = json.selected || false;
    station.createdAt = json.createdAt || Date.now();
    station.imageSize = json.imageSize || 0;
    return station;
  }

  /**
   * Convert to JSON object
   */
  toJSON(): any {
    return {
      id: this.id,
      name: this.name,
      streamUri: this.streamUri,
      imageUrl: this.imageUrl,
      playlistContent: this.playlistContent,
      metadata: this.metadata,
      mimeType: this.mimeType,
      channelCount: this.channelCount,
      sampleRate: this.sampleRate,
      bitrate: this.bitrate,
      playbackState: this.playbackState,
      selected: this.selected,
      createdAt: this.createdAt,
      imageSize: this.imageSize
    };
  }

  /**
   * Clone the station
   */
  clone(): Station {
    const clone = new Station();
    clone.id = this.id;
    clone.name = this.name;
    clone.streamUri = this.streamUri;
    clone.imageUrl = this.imageUrl;
    clone.playlistContent = this.playlistContent;
    clone.metadata = this.metadata;
    clone.mimeType = this.mimeType;
    clone.channelCount = this.channelCount;
    clone.sampleRate = this.sampleRate;
    clone.bitrate = this.bitrate;
    clone.playbackState = this.playbackState;
    clone.selected = this.selected;
    clone.createdAt = this.createdAt;
    clone.imageSize = this.imageSize;
    return clone;
  }
}

/**
 * Station collection manager
 */
export class StationCollection {
  private static instance: StationCollection;
  private stations: Station[] = [];

  private constructor() {
    this.loadFromStorage();
  }

  static getInstance(): StationCollection {
    if (!StationCollection.instance) {
      StationCollection.instance = new StationCollection();
    }
    return StationCollection.instance;
  }

  /**
   * Load stations from local storage
   */
  loadFromStorage(): void {
    try {
      const stored = AppStorage.get<Station[]>('stations');
      if (stored) {
        this.stations = stored.map(s => Station.fromJSON(s));
      }
    } catch (e) {
      console.error('Failed to load stations:', e);
      this.stations = [];
    }
  }

  /**
   * Save stations to local storage
   */
  saveToStorage(): void {
    try {
      AppStorage.setOrCreate('stations', this.stations.map(s => s.toJSON()));
    } catch (e) {
      console.error('Failed to save stations:', e);
    }
  }

  /**
   * Get all stations
   */
  getStations(): Station[] {
    return this.stations;
  }

  /**
   * Add a new station
   */
  addStation(station: Station): boolean {
    if (this.findByUri(station.streamUri)) {
      return false;
    }
    this.stations.push(station);
    this.saveToStorage();
    return true;
  }

  /**
   * Remove a station
   */
  removeStation(station: Station): void {
    const index = this.stations.findIndex(s => s.id === station.id);
    if (index !== -1) {
      this.stations.splice(index, 1);
      this.saveToStorage();
    }
  }

  /**
   * Update a station
   */
  updateStation(station: Station): void {
    const index = this.stations.findIndex(s => s.id === station.id);
    if (index !== -1) {
      this.stations[index] = station;
      this.saveToStorage();
    }
  }

  /**
   * Find station by URI
   */
  findByUri(uri: string): Station | undefined {
    return this.stations.find(s => s.streamUri === uri);
  }

  /**
   * Find station by ID
   */
  findById(id: string): Station | undefined {
    return this.stations.find(s => s.id === id);
  }

  /**
   * Get station count
   */
  getCount(): number {
    return this.stations.length;
  }
}
